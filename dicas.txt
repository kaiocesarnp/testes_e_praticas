PROGRAMAR PYTHON PELO CMD:

instalar o python pela store;
criar o diretório no github;
instalar o git portable e abra o git-bash;
clonar o diretório remoto para o local;

criar arquivo:
	echo.>nome_do_arquivo.py

editar arquivo:
	notepad nome_do_arquivo.py
			isso abrirá o arquivo py num bloco de notas
